Step 2: Implement intersection shader

[shader("intersection")]
void SDF_Intersect()
{
  IntersectionAttrib attr = (IntersectionAttrib)0;
  Ray ray = GetRayInLocalSpace();
  
  const float threshold = 0.0001;
  float t = RayTMin();

  // Do sphere tracing through the AABB.
  UINT i = 0;
  while (i++ < MAX_STEPS && t <= RayTCurrent())
  {
      float3 position = ray.origin + t * ray.direction;
      float distance = GetSDFDistance(position, sdf);

      if (distance <= threshold * t)
      {
          float thit = t;
      /** attr = FillIntersectionAttributes(); */
          ReportHit(thit, /*hitKind*/ 0, attr);
          return;
      }

      t += distance;
  }
}
